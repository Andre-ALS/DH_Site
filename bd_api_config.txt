TABLE user {
   id_user int pk auto increment;
   st_name varchar(250);
   st_email varchar(250) unique;
   st_password varchar(250);
}

TABLE post {
   id_post int pk auto increment;
   st_title varchar(250);
   st_url_photo varchar(250);
   fk_user int;
}

TABLE post_comment {
   id_post_comment int pk auto increment;
   st_comment varchar(250);
   fk_post int;
   fk_user int;
}
